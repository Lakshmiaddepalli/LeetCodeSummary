import java.io.*;
import java.util.*;

class MyCode {
  public static int calculateCost(String[][] requirements, String[][] costs){
    Map<String, List<Node>> graph = new HashMap<>();
    for (String[] req: requirements) {
      String from = req[1];
      if (!graph.containsKey(from)) {
        graph.put(from, new ArrayList<Node>());
      }
      graph.get(from).add(new Node(req[0], Integer.parseInt(req[2])));
    }
    
    Map<String, Integer> costMap = new HashMap<>();
    for(String[] cost: costs) {
      costMap.put(cost[0], Integer.parseInt(cost[1]));
    }
    
    int tractorCost = meterialCost("Tractor", graph, costMap);
    int truckCost = meterialCost("Truck", graph, costMap);
    
    return 500 * tractorCost + 750 * truckCost;
  }
  
  private static int meterialCost(String curr, Map<String, List<Node>> graph, Map<String, Integer> costMap) {
    if (costMap.containsKey(curr)) return costMap.get(curr);
    
    int cost = 0;
    for (Node child: graph.get(curr)) {
      String meterial = child.meterial;
      int count = child.count;
      cost += count * meterialCost(meterial, graph, costMap);
    }
    return cost;
  }
  
  static class Node {
    public String meterial;
    public int count;
    
    public Node (String meterial, int count) {
      this.meterial = meterial;
      this.count = count;
    }
  }
    
  public static void main(String[] args) {
    String[][] requirements = {{"Paint", "Tractor", "3"}, {"Iron", "Plow", "4"}, {"Plow", "Tractor", "1"}, {"Iron", "Engine", "2"}, {"Chemicals", "Engine", "1"}, {"Engine", "Tractor", "1"}, {"Chemicals", "Wheels", "5"}, {"Wheels", "Truck", "18"}, {"Engine", "Truck", "1"}, {"Rubber", "Wheels", "7"}};
    String[][] costs = {{"Paint", "100"}, {"Iron", "900"}, {"Chemicals", "350"}, {"Rubber", "500"}};
    System.out.println(calculateCost(requirements, costs));
  }
}
